\newcommand{\burna}{{\dg Burnside's Lemma} {\gray (Cauchy 1845, Frobenius 1887)}
\\The number of orbits of a permutation group $\bl G$ acting on a set $\bl X$ is\vspace*{-2mm}
\[\bl
  \frac{1}{|G|}\sum_{g\in G}\phi(g)\vspace*{-3mm}
\]
where $\bl\phi(g)\mdef |\{x\myin X\,:\,g(x)\meq x\}|$ is the number of {\dg fixed points} of $\bl g$.}

\newcommand{\burnaexiva}{\\[3mm]\exercise}
\newcommand{\burnaexivb}{\\Determine the number of bead strings of length $\bl n$ using beads in $\bl r$ colours.}

\newcommand{\burnapfia}{\\[3mm]\proof}
\newcommand{\burnapfib}{\\For each $\bl x\myin X$,
let $\bl O_x\meq \{g(x)     : g\myin G\}$
and $\bl S_x\meq \{g\myin G : g(x)\meq x\}$
be the orbit and stabiliser of~$\bl x$.
Let $\bl \mathcal{O}$ be the set of orbits of $\bl G$.}
\newcommand{\burnapfic}{\\By the {\dg Orbit-Stabiliser Theorem}, $\bl |G|\meq |S_x||O_x|$.}
\newcommand{\burnapfid}[9]{\\Hence,\vspace*{-3mm}
\begin{align*}\bl
  |\mathcal{O}|
       &\meq \sum_{O\in\mathcal{O}} 1
  \;{#1{\meq \sum_{O\in\mathcal{O}} \frac{|O|}{|O|}}}
  \;{#2{\meq \sum_{O\in\mathcal{O}} \sum_{x\in O}\frac{1}{|O|}}}
  \;{#3{\meq \sum_{x\in X} \sum_{\substack{O\in\mathcal{O}:\\O = O_x}} \frac{1}{|O|}}}
  \;{#4{\meq \sum_{x\in X} \frac{1}{|O_x|}}}\\
 &\;{#5{\meq \frac{1}{|G|} \sum_{x\in X} |S_x|}}
  \;{#6{\meq \frac{1}{|G|} \sum_{x\in X} \sum_{\substack{g\in G :\\g(x) = x}} \hspace*{-1mm} 1}}
  \;{#7{\meq \frac{1}{|G|} \sum_{g\in G} \sum_{\substack{x\in X :\\g(x) = x}} \hspace*{-1mm} 1}}
  \;{#8{\meq \frac{1}{|G|} \sum_{g\in G} \phi(g)\black\,.\;\Box}}\\[-7mm]
\end{align*}}

\newcommand{\polyaba}{\theorem}
\newcommand{\polyabb}{\\Let $\bl c_k(G)$ be the number of permutations in $\bl G$ with exactly $\bl k$ cycles.
  \\Then the number of orbits of $\bl G$ on $\bl B^A$ is \vspace*{-2mm}
  \[\bl
  \frac{1}{|G|}\sum_{k=1}^\infty c_k(G) |B|^k\,.\]}

\newcommand{\polyabpfa}{\proof}
\newcommand{\polyabpfb}{\\A fixed point of $\bl\sigma\myin G$ is a colouring $\bl f\myin B^A$ for which $\bl \sigma(f)\meq f$;
 \\that is $\bl f(a)\meq f(\sigma^{-1}(a))$ for all $\bl a\myin A$.}
\newcommand{\polyabpfc}{\\Hence, $\bl f$ is constant on each cycle $\bl\sigma_i$ of $\bl\sigma$.}
\newcommand{\polyabpfd}{\\Conversely, if $\bl f$ is constant on each cycle~$\bl\sigma_i$,
then $\bl f$ is a fixed point for~$\bl\sigma$.}
\newcommand{\polyabpfe}{\\Hence if $\bl \sigma$ has $\bl k$ cycles,
then the number of fixed points for $\bl\sigma$ is $\bl|B|^k$.}
\newcommand{\polyabpff}[2]{\\By {\dg Burnside's Lemma}, the number of orbits is thus\vspace*{-2mm}
\[\bl
       \frac{1}{|G|} \sum_{\sigma\in G} \phi(x)
  \meq \frac{1}{|G|} \sum_{k=1}^\infty\hspace*{-5mm}\sum_{\substack{\sigma\in G:\\\text{$\sigma$ has $k$ cycles}}} \hspace*{-5mm}\phi(\sigma)
  \;{#1{\meq \frac{1}{|G|} \sum_{k=1}^\infty\hspace*{-5mm}\sum_{\substack{\sigma\in G:\\\text{$\sigma$ has $k$ cycles}}} \hspace*{-5mm}|B|^k}}
  \;{#2{\meq \frac{1}{|G|} \sum_{k=1}^\infty c_k(G) |B|^k\black\,.\quad\vspace*{-8mm}}}
\]{#2{\qed}}}

\newcommand{\cycinddexaia}{\example}
\newcommand{\cycinddexaib}{\\Let $\bl G$ be the cyclic group $\bl C_n$.}
\newcommand{\cycinddexaic}{\\For each divisor $\bl d|n$, there are $\bl \phi(d)$ permutations of order $\bl d$,
 \\and each of these has $\bl \frac{n}{d}$ cycles of length $\bl d$.}
\newcommand{\cycinddexaid}{\\ Thus, the {\dg cycle index} of $\bl C_n$ is
   \[\bl
    Z_G(X_1,\ldots,X_n) \meq \frac{1}{n} \sum_{d|n} \phi(d) X_d^{n/d}\,.
   \]}

\newcommand{\polyabc}{This may be re-expressed as follows:}
\newcommand{\polyabav}{\\[4mm]{\dg P\'olya's Theorem} {\gray(simple)}}
\newcommand{\polyabavv}{{\dg P\'olya's Theorem} {\gray(simple)}}
\newcommand{\polyabbv}{\\The number of orbits of $\bl G$ on $\bl B^A$ is \vspace*{-2mm}
  \[\bl
    Z_G(|B|,\ldots,|B|) \meq \frac{1}{|G|} \sum_{\sigma\in G} |B|^{z_1(\sigma)+\cdots+z_n(\sigma)}
  \]}
\newcommand{\polyabba}{{\gray This result can be generalised significantly.}}

\newcommand{\polyacexaia}{\example}
\newcommand{\polyacexaib}{\\Consider bracelets with $\bl n$ beads in $\bl b$ colours
  \\identical under rotational symmetry but distinct under reflections.}
%\newcommand{\polyacexaib}{\\Consider bracelets with $\bl n$ black or white beads,
%  \\identical under rotational symmetry but distinct under reflections.}
\newcommand{\polyacexaic}{\\Then $\bl G$ is the cyclic group $\bl C_n$ and $\bl |B|\meq b$.}
\newcommand{\polyacexaid}{\\By {\dg P\'olya's Theorem}, the number of distinct bracelets with $\bl n$ beads is\vspace*{-2mm}
   \[\bl
    Z_G(b,\ldots,b) \meq \frac{1}{n} \sum_{d|n} \phi(d) b^{n/d}\,.\vspace*{-2mm}
   \]}
\newcommand{\polyacexaie}{For $\bl n\meq 6$ black or white beads ($\bl b\meq 2$), we have {\bl 14} distinct bracelets.}
\newcommand{\polyacexaiev}[1]{For $\bl n\meq 6$,\vspace*{-3mm}
  \begin{align*}\bl
                Z_G(b,\ldots,b)
    &\;{  {\meq \frac{1}{6} \bigl(\phi(1) b^{6/1}+\phi(2) b^{6/2}+\phi(3) b^{6/3}+\phi(6) b^{6/6}\bigr)}}\\
    &\;{#1{\meq \frac{1}{6} \bigl(b^6 + b^3 + 2 b^{2} + 2b\bigr)}}\\[-15mm]
  \end{align*}}
\newcommand{\polyacexaieva}{For instance, the number of bracelets with $\bl 6$ beads in $\bl b\meq 3$ colours is \vspace*{-2mm}
  \[\bl
    \frac{1}{6} \bigl(3^6 + 3^3 + 2\cdot3^{2} + 2\cdot3\bigr)
    \meq 130\black\,.\vspace*{-1mm}\]}
\newcommand{\polyacexaievb}{As we have seen, the number of bracelets with $\bl 6$ beads in $\bl b\meq 2$ colours~is\vspace*{-2mm}
  \[\bl
    \frac{1}{6} \bigl(2^6 + 2^3 + 2\cdot2^{2} + 2\cdot2\bigr)
    \meq 14\black\,.\vspace*{-1mm}\]}
