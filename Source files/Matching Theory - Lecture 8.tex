\newcommand{\lectviiiga}{{\darkgreen Theorem}}
\newcommand{\lectviiigb}{\\The sets of axioms (I1-3), (B1-2), and (C1-3) are equivalent.}
\newcommand{\lectviiigc}[1]{\\In particular, if
  \\[.75mm]\mybullet if $\bl (E,\mathcal{I})$ is a matroid, then its\\
    \hspace*{6.6mm}{\dg bases} $\bl \mathcal{B}$,
      the maximal members of $\bl \mathcal{I}$, satisfy (B1-2)\\
    \hspace*{6.6mm}{\dg circuits} $\bl \mathcal{C}$,
      the minimal subsets not in {$\bl\mathcal{I}$}, satisfy (C1-3)#1}
\newcommand{\lectviiigca}{\\[.6mm]\mybullet if $\bl (E,\mathcal{B})$ is a matroid, then its\\
    \hspace*{6.6mm}{\dg independent sets} $\bl \mathcal{I}$,
      the subsets of $\bl \mathcal{B}$'s members, satisfy (I1-3)\\
    \hspace*{6.6mm}{\dg circuits} $\bl \mathcal{C}$,
      the minimal non-subsets of $\bl \mathcal{B}$'s members, satisfy (C1-3)}
\newcommand{\lectviiigcb}{\\[.6mm]\mybullet if $\bl (E,\mathcal{C})$ is a matroid, then its\\
    \hspace*{6.6mm}{\dg independent sets} $\bl \mathcal{I}$,
      the proper subsets of $\bl \mathcal{C}$'s members, satisfy (I1-3)\\
    \hspace*{6.6mm}{\dg bases} $\bl \mathcal{B}$,
      the maximal proper subsets of $\bl \mathcal{C}$'s members, satisfy (B1-2).}

\newcommand{\lectviiigpfa}{\proof\quad}
\newcommand{\lectviiigpfb}{We will merely prove that (C3) follows from (I1-3).}
\newcommand{\lectviiigpfc}{\\Let $\bl (E,\mathcal{I})$ be a matroid and let $\bl \mathcal{C}$ be the minimal subsets not in $\bl \mathcal{I}$.}
\newcommand{\lectviiigpfd}{\\Suppose that $\bl C_1,C_2\myin \mathcal{C}$ are distinct and that $\bl e\myin C_1\cap C_2$.}
\newcommand{\lectviiigpfe}{\mypicture{
  \rput(-4.882,-5.45){  (C3) \: If $\bl C_1,C_2\myin \mathcal{C}$ are distinct and $\bl e\myin C_1\cap C_2$,}
  \rput(-4.882,-5.95){\hspace*{11mm}then $\bl C\msubseteq (C_1\cup C_2) - e$ for some $\bl C\myin \mathcal{C}$.}}}
\newcommand{\lectviiigpff}{\\Assume that $\bl (C_1\cup C_2) - e\myin \mathcal{I}$.}
\newcommand{\lectviiigpfg}{\\By construction, $\bl C_2-C_1\mneq\emptyset$, so we may choose some $\bl f\myin C_2-C_1$.}
\newcommand{\lectviiigpfh}{\\Now, $\bl C_2$ is minimally not in $\bl \mathcal{I}$, so $\bl C_2-f\myin \mathcal{I}$.}
\newcommand{\lectviiigpfj}{\\Choose $\bl I\myin \mathcal{I}$ to be maximal so that $\bl C_2-f\msubseteq I\msubseteq C_1\cup C_2$.}
\newcommand{\lectviiigpfk}{\\Note that  $\bl C_2\mnsubseteq I$; therefore, $\bl f\mnotin I$.}
\newcommand{\lectviiigpfl}{\\Similarly, $\bl C_1\mnsubseteq I$, so we may choose some $\bl g\myin C_1-I$.}
\newcommand{\lectviiigpfm}{\\Since $\bl f\mynotin C_1$, the elements $\bl f$ and $\bl g$ are distinct}
\newcommand{\lectviiigpfn}{, so\vspace*{-1mm}
  \[\bl|I| \mleq |(C_1\cup C_2) - \{f,g\}| \meq |C_1\cup C_2| -2 \mlt  |(C_1\cup C_2) - e|\,.\]}
\newcommand{\lectviiigpfo}{By (I3), $\bl I\cup\{a\}\myin \mathcal{I}$ for some $\bl a\myin (C_1\cup C_2) - e$ not in $\bl I$.}
\newcommand{\lectviiigpfp}{\\However, this contradicts the maximality of $\bl I$.}
\newcommand{\lectviiigpfq}{\\Hence, $\bl (C_1\cup C_2) - e$ is not in $\bl \mathcal{I}$ and thus contains some $\bl C\msubseteq\mathcal{C}$.\qed}

\newcommand{\lectviiij}{Let $\bl G\meq (V,E)$ be an edge-weighted connected graph.}
\newcommand{\lectviiija}{{\dg\sc Greedy Algorithm} {\gray (Kruskal 1956)}}
\newcommand{\lectviiijb}{\\[1mm]\mybullet Set $\bl T =\emptyset$.}
\newcommand{\lectviiijc}{\\\mybullet Let $\bl A$ be the edges in $\bl E-E(T)$ that form no cycle with edges~in~$\bl T$.}
\newcommand{\lectviiijd}{\\\mybullet Choose an edge $\bl e$ from $\bl A$ that has smallest weight.}
\newcommand{\lectviiije}{\\\mybullet Add $\bl e$ to $\bl T$.}
\newcommand{\lectviiijf}{\\\mybullet Continue until $\bl T$ is a minimal spanning tree for $\bl G$.}

\newcommand{\lectviiik}{Let $\bl M\meq (E,\mathcal{I})$ be a matroid with weights $\bl w(e)$ on each element $\bl e\myin E$.}
\newcommand{\lectviiika}{{\dg\sc Greedy Algorithm For Matroids} {\gray (Borůvka 1926)}}
\newcommand{\lectviiikb}{\\[1mm]\mybullet Set $\bl B =\emptyset$.}
\newcommand{\lectviiikc}{\\\mybullet Let $\bl A$ be all elements $\bl e$ in $\bl E-B$ for which $\bl B\cup \{e\}\myin \mathcal{I}$.}
\newcommand{\lectviiikd}{\\\mybullet Choose an edge $\bl e$ from $\bl A$ with smallest weight $\bl w(e)$.}
\newcommand{\lectviiike}{\\\mybullet Add $\bl e$ to $\bl B$.}
\newcommand{\lectviiikf}{\\\mybullet Continue until $\bl A \meq \emptyset$.}
\newcommand{\lectviiikg}{\\\mybullet Then $\bl B$ is a minimally weighted basis for $\bl M$.}

\newcommand{\lectviiikpfa}{\proof\quad}
\newcommand{\lectviiikpfb}{By (I3), the algorithm will return a basis $\bl B \meq \{e_1,\ldots,e_r\}$.}
\newcommand{\lectviiikpfc}{\\Assume that $\bl w(B') \mlt  w(B)$ for some basis $\bl B' \meq \{f_1,\ldots,f_r\}$ of $\bl M$.}
\newcommand{\lectviiikpfd}{\\We may suppose that
  $\bl w(e_1)\mleq \cdots\mleq w(e_r)$ and
  $\bl w(f_1)\mleq \cdots\mleq w(f_r)$.}
\newcommand{\lectviiikpfe}{\\Set $\bl i \meq \min \{ j : w(f_j) \mlt  w(e_j)\}$. }
\newcommand{\lectviiikpff}{\\By (I3), there is some $\bl j\myin \{1,\ldots,i\}$
  for which $\bl \{e_1,\ldots,e_{i-1}\}\cup\{f_j\}\myin \mathcal{I}$. }
\newcommand{\lectviiikpfg}{\\Also, $\bl w(f_j)\mleq w(f_i) \mlt  w(e_i)$. }
\newcommand{\lectviiikpfh}{However, this means that the algorithm would not have chosen $\bl e_i$
  at the $\bl i$th step for $\bl B$, a contradiction. }
\newcommand{\lectviiikpfj}{\\Thus, $\bl B$ is a basis with minimal weight.\qed}

\newcommand{\lectviiin}{Let $\bl M\meq (E,r)$ be a matroid on $\bl n \meq |E|$ elements.}
\newcommand{\lectviiina}{\\[2mm]For all $\bl A\msubseteq E$, define\vspace*{-2mm}
  \[\blue r^*(A) :\meq |A| - r(E) + r(E-A)\vspace*{-6mm}\]}
\newcommand{\lectviiinav}{\\For all $\bl A\msubseteq E$, define $\bl r^*(A) :\meq |A| - r(E) + r(E-A)$.}
\newcommand{\lectviiinab}{\\[2mm]It is easy to show that $\bl M^*\mdef (E,r^*)$ is also a matroid {\gray (the {\green dual} of $M$)}.}
\newcommand{\lectviiinb}{\\Also define, for $\bl k\black\meq r(E)$ and all $\bl i,j$,\vspace*{-3mm}
  \[\begin{array}{rlrl}\blue
      f_i   &\meq \max\{\,|A|: r(A)   \meq i\}\qquad\quad &\blue U \black &\meq \{f_0+1,\ldots,f_{k-1}+1\}\\\blue
      f^*_j &\meq \max\{\,|A|: r^*(A) \meq j\}            &\blue V \black &\meq \{n-f^*_{n-k-1},\ldots,n-f^*_0\}
    \end{array}\]}
\newcommand{\lectviiinc}{{\darkgreen Theorem {\gray (Britz et al.~2012)}}\vspace*{-2mm}
  \[\blue U\cup V \meq \{1,\ldots,n\}\qquad\mytext{and}\qquad
          U\cap V \meq \emptyset\vspace*{-8mm}\]}

\newcommand{\lectviiincpfa}{\proof\quad}
\newcommand{\lectviiincpfb}{Assume that the theorem is false.}
\newcommand{\lectviiincpfc}{\\Then $\bl f_i+1 \meq n-f^*_j$ for some $\bl i,j$.}
\newcommand{\lectviiincpfd}{\\Choose $\bl A\msubseteq E$ so that $\bl |A| \meq f_j^*$ and $\bl r^*(A) \meq j$.}
\newcommand{\lectviiincpfe}{\\Then $\bl |E-A| \meq n - |A| \meq n - f_j^* \meq f_i + 1$, so $\bl r(E-A)\mgeq i+1$.}
\newcommand{\lectviiincpff}{\\By definition, $\bl -|A| + r^*(A) + r(E) \meq r(E-A)$;
  \\therefore,\hspace*{2mm}            $\bl -f_j^*+j+k\mgeq i+1$.}
\newcommand{\lectviiincpfg}{\\Similarly, $\bl n-f_i  +i-k\mgeq j+1$.}
\newcommand{\lectviiincpfh}{\\Adding these inequalities gives $\bl 1=n-f_i-f_j^*\mgeq 2$, a contradiction.\qed\vspace*{-10mm}}
