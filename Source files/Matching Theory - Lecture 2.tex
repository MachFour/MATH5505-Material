\newcommand{\hallime}{\\[3mm]{\darkgreen\sf Theorem}\\

\newcommand{\halliicd}{{\darkgreen Dilworth's Theorem} {\gray(1950)}\\
  The minimal number $\bl m$ of disjoint chains covering a finite poset $\bl P$ {\em\dg equals}
  \\the maximal size $\bl M$ of an antichain.}

\newcommand{\halliipfiiia}{\\[2.5mm]\proof}
\newcommand{\halliipfiiib}{Note that $\bl m\mgeq M$. }
\newcommand{\halliipfiiic}{We therefore only need to prove that $\bl m\mleq M$.}
\newcommand{\halliipfiiid}{\\Assume that the theorem holds for all posets on fewer than $\bl |P|$ elements. }
\newcommand{\halliipfiiie}{\\Choose a maximal chain $\bl C$ in $\bl P$. }
\newcommand{\halliipfiiif}{\\If every antichain in $\bl P-C$ has at most $\bl M-1$ elements,
  then by the induction assumption, $\bl P-C$ can be covered by $\bl M-1$ disjoint chains.}
\newcommand{\halliipfiiig}{\\Together with $\bl C$, these chains cover $\bl P$. }
\newcommand{\halliipfiiih}{Hence, $\bl m\mleq M$. }
\newcommand{\halliipfiiij}{\\Suppose that $\bl A$ is an antichain in $\bl P-C$ with $\bl M$ elements. }
\newcommand{\halliipfiiik}{\\Set\vspace{-7.5mm}
  \begin{align*}\bl
    A^{-} &\meq \{x\myin P\::\: x \black\mpreceq\bl a\:\,\mytext{for some}\,\:a\myin A\}\hspace{3cm}\\[-.5mm]\bl
    A^{+} &\meq \{x\myin P\::\: x \black\succeq\bl a\:\,\mytext{for some}\,\:a\myin A\}\end{align*}}
\newcommand{\halliipfiiim}{\\[-8mm]Since $\bl |A| \meq  M$, it follows that $\bl P \meq  A^{-}\cup A^{+}$. }
\newcommand{\halliipfiiin}{\\By the maximality of $\bl C$, the element $\bl \max(C)$ is not in $\bl A^{-}$. }
\newcommand{\halliipfiiio}{\\Hence, $\bl |A^{-}|\mlt |P|$. }
\newcommand{\halliipfiiip}{The induction assumption implies
  \\that $\bl A^{-}$ is a union of $\bl M$ disjoint chains $\bl C^{-}_a$ with $\bl \max(C^{-}_a)\meq a\myin A$.}
\newcommand{\halliipfiiiq}{\\Similarly, $\bl A^{+}$ is a union of $\bl M$ disjoint chains $\bl C^{+}_a$ with $\bl \min(C^{+}_a)\meq a\myin A$.}
\newcommand{\halliipfiiir}{\\Then $\bl P$ is the union of the $\bl M$ disjoint chains $\bl C^{-}_a\cup C^{+}_a$.}
\newcommand{\halliipfiiis}{\\Hence, $\bl m\mleq M$.\qed}
