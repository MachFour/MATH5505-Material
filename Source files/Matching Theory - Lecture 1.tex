\newcommand{\hallic}{\\A {\em\dg matching} of $\bl\mathcal{A}$ is
  a set of distinct elements $\bl\{a_i\}_I$ with $\bl a_i\myin A_i$ ($\bl\forall i\myin I$).}
\newcommand{\hallid}{\\[3mm]\example}
\newcommand{\hallida}{\\$\bl {\blue\mathcal{A}} \meq  \{A_1,A_2,A_4\}$ where $\bl A_i$ is the set of boys that girl $\bl i$ likes.}
\newcommand{\hallidb}{\\Then $\bl \{\textrm{boy \blue1},\textrm{boy \blue2},\textrm{boy \blue3}\}$ is a {\blue matching} of $\bl\mathcal{A}$.}
\newcommand{\hallide}{\\In contrast, $\bl {\blue\mathcal{B}} \meq  \{A_1,A_2,A_3,A_4\}$ has no matching.}
\newcommand{\halliea}{\\Set $\bl {\blue\mathcal{A}(J)} \meq  \bigcup_{i\myin J} A_i$ for all $\bl J\msubseteq I$.}

\newcommand{\hallieb}{\\[4mm]{\darkgreen The Marriage Theorem} \;{\gray (Frobenius 1917, Hall 1935)}\\
  $\bl\mathcal{A}$ has a matching if and only if $\bl|\mathcal{A}(J)|\mgeq |J|$ for all $\bl J\msubseteq I$.}
\newcommand{\halliebv}{{\darkgreen The Marriage Theorem} \;{\gray (Frobenius 1917, Hall 1935)}\\
  $\bl\mathcal{A}$ has a matching if and only if $\bl|\mathcal{A}(J)|\mgeq |J|$ for all $\bl J\msubseteq I$.}
\newcommand{\halliec}{\\[3mm]{\gray Also called ``{\green Hall's Theorem}".}}

\newcommand{\hallipfia}{\\[3mm]\proof}
\newcommand{\hallipfiav}{\\[3mm]\proof {(continued)}}
\newcommand{\hallipfib}{\\If $\bl |\mathcal{A}(J)|\mlt |J|$ for some $\bl J\msubseteq I$, then $\bl \mathcal{A}$ has no matching.}
\newcommand{\hallipfic}{\\Suppose therefore that $\bl|\mathcal{A}(J)|\mgeq |J|$ for all $\bl J\msubseteq I$.}
\newcommand{\hallipfid}{\\If $\bl|A_{i_0}| \mgeq 2$ for some $\bl i_0\myin I$, then let $\bl x$ and $\bl y$ be distinct elements of $\bl A_{i_0}$.}
\newcommand{\hallipfie}{\\Assume that we cannot remove $\bl x$ or $\bl y$ from $\bl A_{i_0}$ without violating the inequality in the theorem. }
\newcommand{\hallipfif}{Then there are subsets $\bl I_x,I_y\msubseteq I-i_0$ so~that\vspace{-1mm}
  \begin{align*}\bl
    |\mathcal{A}(I_x)\cup (A_{i_0}-\{x\})| &\mlt  |I_x|+1\\\bl
    |\mathcal{A}(I_y)\cup (A_{i_0}-\{y\})| &\mlt  |I_y|+1\vspace{-3mm}
  \end{align*}}
\newcommand{\hallipfifv}{Then there are subsets $\bl I_x,I_y\msubseteq I-i_0$ so~that\vspace{-1mm}
  \begin{align*}\bl
    |\mathcal{A}(I_x)\cup (A_{i_0}-\{x\})| &\mleq  |I_x|\\\bl
    |\mathcal{A}(I_y)\cup (A_{i_0}-\{y\})| &\mleq  |I_y|\vspace{-3mm}
  \end{align*}}
\newcommand{\hallipfig}[5]{Thus,\vspace{-1mm}
  \begin{align*}\bl
         \quad |I_x| + |I_y|
    &\:{{\mgeq | \mathcal{A}(I_x)\cup (A_{i_0} - x)|  +  |\mathcal{A}(I_y)\cup (A_{i_0} - y)|}}\\\bl
    &\:{#1{\mgeq | \mathcal{A}(I_x)\cup (A_{i_0} - x)  \cup \mathcal{A}(I_y)\cup (A_{i_0} - y)|
       +  |\mathcal{A}(I_x)\cap \mathcal{A}(I_y)|}}\\\bl
    &\:{#2{\meq |\mathcal{A}(I_x\cup I_y \cup \{i_0\})|
         +|\mathcal{A}(I_x\cap I_y)|}}\\\bl
    &\:{#3{\mgeq |I_x\cup I_y| + 1
         +|I_x\cap I_y|}}\\\bl
    &\:{#4{\meq     |I_x| + |I_y| + 1}}\vspace{-3mm}
  \end{align*}
  {#5{which is a contradiction. \qed}}}
\newcommand{\hallipfigv}{Thus, $\bl|I_x| + |I_y| \mgeq \cdots \mgeq |I_x| + |I_y| + 1$, which is a contradiction.\\}
\newcommand{\hallipfih}{We can thus remove elements from the sets $\bl A_i$ while preserving the
  \\inequality in the theorem,
  until $\bl A_i \meq  \{a_i\}$ for some $\bl a_i$, for each $\bl i\myin I$.}
\newcommand{\hallipfij}{\\Then $\bl|\{a_i\}_I| \meq  |\mathcal{A}(I)| \mgeq |I|$.}
\newcommand{\hallipfik}{\\Hence, the elements $\bl a_i$ are distinct,

\newcommand{\halliha}{{\darkgreen\sf Theorem {\gray (Ore 1955)}}\\
  $\bl\mathcal{A}$ has a partial matching of size $\bl k$ if and only if, for all $\bl J\msubseteq I$,\\
  $\bl|\mathcal{A}(J)|\mgeq |J| - |I| + k$.}
\newcommand{\hallihb}{\\[2mm]{\darkgreen\sf Theorem \gray(Ore 1955)}\\
  A set $\bl B$ contains a partial matching of $\bl\mathcal{A}$ of size $\bl k$ {\em iff},
  for all $\bl J\msubseteq I$, $\bl|\mathcal{A}(J)\cap B|\mgeq |J| - |I| + k$.}
\newcommand{\hallihc}{\\[2mm]{\darkgreen\sf Theorem \gray(Ore 1955)}\\
  A set $\bl B$ is a partial matching of $\bl\mathcal{A}$ if and only if, for all $\bl J\msubseteq I$,
  \\$\bl |\bl\mathcal{A}(J)\cap B|\mgeq |J| - |I| + |B|$.}
\newcommand{\hallihd}{\\[2mm]{\darkgreen\sf Theorem}\\
  A set $\bl B$ is a partial matching of $\bl\mathcal{A}$ if and only if, for all $\bl A\msubseteq B$,
  \\$\bl|\{i\myin I\::\:A_i\cap A\neq \emptyset\}|\mgeq|A|$.}

\newcommand{\hallipfiia}{\\[3mm]\proof}
\newcommand{\hallipfiib}{\\Let $\bl D$ be a set of $\bl|I|-k$ elements not in $\bl \mathcal{A}(I)$. }
\newcommand{\hallipfiic}{Set $\bl \mathcal{B} \meq  \{ A_i \cup D\}_I$. }
\newcommand{\hallipfiid}{\\Then $\bl \mathcal{A}$ has a partial matching of size $\bl k$ if and only if $\bl \mathcal{B}$ has a matching. }
\newcommand{\hallipfiie}{\\For each $\bl J\msubseteq I$,\vspace{-3mm}
  \[\bl
    |\mathcal{B}(J)| \meq  \bigl|\bigcup_{i\myin J}A_i\cup D\bigr| \meq  |\mathcal{A}(J)| + |D| \meq  |\mathcal{A}(J)| + |I| - k\vspace{-3mm}
  \]}
\newcommand{\hallipfiif}{By the {\dg Marriage Theorem}, $\bl \mathcal{B}$ has a matching if and only if $\bl|\mathcal{B}(J)| \mgeq |J|$ for all $\bl J\msubseteq I$. }
\newcommand{\hallipfiig}{In other words, $\bl \mathcal{A}$ has a partial matching of size $\bl k$ if and only if
$\bl|\mathcal{A}(J)| \mgeq |J| - |I| + k$ for all $\bl J\msubseteq I$. \qed\vspace{-12mm}}

\newcommand{\hallimd}{{\darkgreen Lemma}\\
  $\bl B\myin\mathcal{I}(\mathcal{A})$  {\em\dg iff}\; the rows of $\bl {\blue M_{\mathcal{A}}}$ indexed by $\bl B$ are linearly independent.}

\newcommand{\hallipfvia}{\\[2.5mm]\proof}
\newcommand{\hallipfvib}{The rows of $\bl M_{\mathcal{A}}$ indexed by $\bl B$ are linearly independent
  if~and~only~if\\they contain a non-singular square sub-matrix $\bl N$ of order~$\bl|B|$. }
\newcommand{\hallipfvic}{\\Such $\bl N$ is non-singular precisely when $\bl \det N \meq  \cdots \pm \prod_{a\myin B} z_{ai_a} + \cdots \neq 0$. }
\newcommand{\hallipfvid}{This happens exactly when there is at least one non-zero term $\bl \prod_{a\myin B} z_{ai_a}$
  (since the $\bl z_{ai_a}$ are independent variables and cannot cancel each other).}
\newcommand{\hallipfvie}{\\This, in turn, occurs when $\bl B$ is a partial matching.\qed}

\newcommand{\hallime}{\\[3mm]{\darkgreen\sf Theorem}\\
  The family $\bl\mathcal{I}(\mathcal{A})$ satisfies the following properties:
  \\[.5mm]\mybullet $\bl \emptyset\myin\mathcal{I}(\mathcal{A})$;
  \\[.5mm]\mybullet If $\bl A\myin\mathcal{I}(\mathcal{A})$   and $\bl B\msubseteq A$, then $\bl B\myin\mathcal{I}(\mathcal{A})$;
  \\[.5mm]\mybullet If $\bl A,B\!\myin\!\mathcal{I}(\mathcal{A})$ and $\bl|A|\!\mlt \!|B|$,
    then $\bl A\cup\{a\}\!\myin\!\mathcal{I}(\mathcal{A})$ for some $\bl a\!\myin\!B\!-\!A$.}

\newcommand{\hallimf}{\\[3mm]Since the tuple $\bl(E,\mathcal{I}(\mathcal{A}))$ satisfies these properties,
  it is a {\em\dg matroid}\\(or {\em\dg independence structure}, or {\em\dg combinatorial geometry}).}
\newcommand{\hallimfv}{\\[3mm]Since the tuple $\bl(E,\mathcal{I}(\mathcal{A}))$ satisfies these properties,
  it is a {\em\dg  matroid}.}
\newcommand{\hallimg}{\\Matroids arise in numerous ways and are very useful in combinatorics.}
