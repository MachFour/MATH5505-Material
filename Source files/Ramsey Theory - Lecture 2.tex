\newcommand{\ramseybexda}{\\[6mm]\lemma {\gray (Erd\H{o}s \&\ Szekeres 1935)}}
\newcommand{\ramseybexdav}{\lemma}
\newcommand{\ramseybexdb}{\\[.5mm]$\bl R(n,\ell)\mleq R(n-1,\ell) + R(n,\ell-1)$}

\newcommand{\ramseybexdaexaa}{\\[6mm]\example}
\newcommand{\ramseybexdaexab}{\\[.5mm]For any $\bl n\geq 2$, we note that $\bl R(n,2) \meq R(2,n)\meq n$. {\gray (Why?)}}
\newcommand{\ramseybexdaexac}{\\The above bound is therefore tight for $\bl R(3,3)$:
   \[\bl R(3,3) \meq 6 \mleq 3 + 3 \meq R(2,3) + R(3,2)\,\black.\vspace*{-6mm}\]}

\newcommand{\ramseybexdpfa}{\\[3mm]\proof}
\newcommand{\ramseybexdpfb}{\\We use induction on $\bl n+\ell$}
\newcommand{\ramseybexdpfc}{, noting that $\bl R(n,1) \meq R(1,n) \meq 1$ for all $\bl n$.}
\newcommand{\ramseybexdpfd}{\\Assume that $\bl R(n-1,\ell)$ and $\bl R(n,\ell-1)$ are well-defined\hfill}
\newcommand{\ramseybexdpfe}{\\and set $\bl N\black := \bl R(n-1,\ell) + R(n,\ell-1)$.}
\newcommand{\ramseybexdpff}{\\Colour each edge of $\bl K_N$ either {\red red} or {\bl blue} and choose a vertex $\bl v$.}
\newcommand{\ramseybexdpfg}{\\Let $\red R$ and $\bl B$ be the vertices with {\red red} and {\bl blue} edges to $\bl v$, respectively.}
\newcommand{\ramseybexdpfh}[1]{\\Then\vspace*{-2mm}
  \[\bl
    |\red R\bl| + |B| \meq N - 1 \;{#1{\meq R(n-1,\ell) + R(n,\ell-1) - 1\,\black.}}
  \]}
\newcommand{\ramseybexdpfj}{Therefore, $\bl|\red R\bl|\mgeq R(n-1,\ell)$ or $\bl |B|\mgeq R(n,\ell-1)$.}
\newcommand{\ramseybexdpfk}{\\Assume without loss of generality that $\bl|\red R\bl|\mgeq R(n-1,\ell)$.}
\newcommand{\ramseybexdpfm}{\\If the vertices $\red R$ induce a {\bl blue}-edged $\blue K_\ell$,
  then we are done.}
\newcommand{\ramseybexdpfn}{\\Otherwise, by definition of $\bl R(n-1,\ell)$,
   $\red R$ induces a {\red red}-edged $\red K_{n-1}$.}
\newcommand{\ramseybexdpfo}{\\Adding the $\bl v$ and its {\red red} edges,
  we have a {\red red}-edged $\red K_n$.\qed}

   If $\bl n_1,\ldots,n_k\myin\mathbb{N}$ and $\bl n$ is sufficiently large,\\
   then each colouring of $\bl \binom{[n]}{2}$ with colours $\bl c_1,\ldots,c_k$\\
   gives a $\bl c_i$-coloured subfamily $\bl \binom{S}{2}$
   for some $\bl i$ and $\bl n_i$-subset $\bl S\subseteq [n]$.}

\newcommand{\ramseydv}{{\dg Ramsey's Theorem} {\gray (1930)} {\lightgray (reduced)}\\
   If $\bl n_1,\ldots,n_k\myin\mathbb{N}$ and $\bl n$ is sufficiently large,\\
   then each colouring of $\bl \binom{[n]}{\red2\bl}$ with colours $\bl c_1,\ldots,c_k$\\
   gives a $\bl c_i$-coloured subfamily $\bl \binom{S}{\red2\bl}$
   for some $\bl i$ and $\bl n_i$-subset $\bl S\subseteq [n]$.}

\newcommand{\ramseye}{{\dg Ramsey's Theorem} {\gray (1930)}\\
   If $\bl n_1,\ldots,n_k,r\myin\mathbb{N}$ and $\bl n$ is sufficiently large,\\
   then each colouring of $\bl \binom{[n]}{r}$ with colours $\bl c_1,\ldots,c_k$\\
   gives a $\bl c_i$-coloured subfamily $\bl \binom{S}{r}$
   for some $\bl i$ and $\bl n_i$-subset $\bl S\subseteq [n]$.}

\newcommand{\ramseyea}{\\[2mm]{\gray The least such $\lightblue n$ is denoted $\lightblue R_r(n_1,\ldots,n_k)$.}}

\newcommand{\ramseyepfa}{\\[6mm]\proof}
\newcommand{\ramseyepfav}{{\phantom{ }\vspace*{-8mm}}\proof}
\newcommand{\ramseyepfb}{\\[.5mm]We generalise {\dg Proof 1}, using induction on $\bl r$ having proved
  the case $\bl r\meq 2$.}
\newcommand{\ramseyepfc}{\\Assume that the theorem is true for $\bl r-1$ and set $\red t\black :=\bl R_{r-1}(n_1,\ldots,n_k)$.}
\newcommand{\ramseyepfd}{\\Choose $\bl r-2$ elements $\bl v_1,\ldots,v_{r-2}$ and
  set $\bl V_{r-1}\black:=\bl [n]-\{v_1,\ldots,v_{r-2}\}$.}
\newcommand{\ramseyepfe}{\\Continue to choose elements $\bl v_{r-1},v_r,\ldots$ and sets $\bl V_r,V_{r+1},\ldots$ as follows.}
\newcommand{\ramseyepff}{\\Suppose that we have chosen $\bl v_1,\ldots,v_{i-1}$ and $\bl V_i$. }
\newcommand{\ramseyepfg}{Choose $\bl v_i\myin V_i$.}
\newcommand{\ramseyepfh}{\\Let $\bl V_{i+1}$ be a maximal subset of $\bl V_i-\{v_i\}$
  so that $\bl T\cup\{v\}$ and $\bl T\cup\{w\}$
  have same colour for all $\bl v,w\in V_{i+1}$
  and $\bl (r-1)$-subsets $\bl T\subseteq\{v_1,\ldots,v_i\}$.}
\newcommand{\ramseyepfj}{\\If $\bl n$ is sufficiently large, then it is possible to choose $\red t$
  elements $\bl v_1,\ldots,v_{\red t}$.}
\newcommand{\ramseyepfk}{\\Now colour each $\bl (r-1)$-subset $\bl T$ of these,
  giving $\bl T$ the same colour as $\bl T\cup \{v_t\}$ if $\bl v_t\notin V_{\red t}$
  and colouring it randomly otherwise.}
\newcommand{\ramseyepfm}{\\By definition of $\red t$,
  there is a $\bl n_i$-subset $\bl S$ of the $\bl v_i$s whose $\bl (r-1)$-subsets
  each have colour $\bl c_i$ for some $\bl i$.}
\newcommand{\ramseyepfn}{\\Let $\bl U\subseteq S$ be an $\bl r$-subset
  and let $\bl v$ be the greatest element in $\bl U$.}
\newcommand{\ramseyepfo}{\\Then $\bl T\black :=\bl U-\{v\}$ is a $\bl c_i$-coloured $\bl (r-1)$-sized subset of $\bl S$}
\newcommand{\ramseyepfp}{\\and $\bl T \cup \{v_t\}$ must therefore be $\bl c_i$-coloured.}
\newcommand{\ramseyepfq}{\\By construction, $\bl T \cup \{v_t\}$ and $\bl U\meq T \cup \{v\}$ have the same colour,~$\bl c_i$.}
\newcommand{\ramseyepfr}{\\We see that each $\bl r$-subset $\bl U\subseteq S$ is $\bl c_i$-coloured. \qed\vspace*{-8mm}}
