\newcommand{\treee}{{\darkgreen\sf Theorem}
  \\The spanning trees of a connected graph have the same number of~edges.}

\newcommand{\treepfiia}{\\[2.5mm]\proof }
\newcommand{\treepfiib}{\\If $\bl G$ is a connected graph on $\bl n$ vertices,
  \\then any spanning tree $\bl T$ of $\bl G$ will contain all $\bl n$ vertices. }
\newcommand{\treepfiic}{\\Hence, $\bl T$ has $\bl n-1$ edges.\qed}

\newcommand{\treef}{{\darkgreen\sf Theorem}
  \\Every connected graph contains a spanning tree.}

\newcommand{\treepfiiia}{\\[2.5mm]\proof }
\newcommand{\treepfiiib}{\\Let $\bl G$ be a connected graph. }
\newcommand{\treepfiiic}{\\If $\bl G$ is a tree, then $\bl G$ is a spanning tree for itself.}
\newcommand{\treepfiiid}{\\If $\bl G$ is not a tree, then $\bl G$ contains at least one cycle. }
\newcommand{\treepfiiie}{\\Remove an edge from a cycle in $\bl G$. }
\newcommand{\treepfiiif}{\\The resulting graph is still connected and contains all vertices of $\bl G$. }
\newcommand{\treepfiiig}{\\Continue to remove an edge from each cycle until no cycles remain. }
\newcommand{\treepfiiih}{\\The resulting graph is a tree that contains all vertices of $\bl G$. }
\newcommand{\treepfiiij}{\\In other words, it is a spanning tree of $\bl G$.\qed}

\newcommand{\treeg}{{\darkgreen\sf Theorem}
  \\A connected graph on  {\blue$n$}   vertices is a {\blue tree}
    if and only if it has {\blue$n-1$} edges.}

\newcommand{\treepfiva}{\\[2mm]\proof }
\newcommand{\treepfivb}{Let $\bl G$ be a connected graph with $\bl n$ vertices. }
\newcommand{\treepfivc}{\\If $\bl G$ is a tree, then it has $\bl n-1$ edges. }
\newcommand{\treepfivd}{\\Conversely, suppose that $\bl G$ has $\bl n-1$ edges. }
\newcommand{\treepfive}{\\Let $\bl T$ be a spanning tree of $\bl G$. }
\newcommand{\treepfivf}{\\Then $\bl T$ is contained in $\bl G$ and has as many vertices and edges as~$\bl G$. }
\newcommand{\treepfivg}{\\Therefore, $\bl T$ is in fact equal to $\bl G$. }
\newcommand{\treepfivh}{\\Hence, $\bl G$ is a tree.\qed}

\newcommand{\treep}{We will prove this in {\darkgreen Lecture 8}.
  \\[5mm]The {\sc\dg Greedy Algorithm} (also called {\sc\dg Kruskal's Algorithm})
    also works for weighted matchings, weighted matroids, and even more general and abstract objects!}

\newcommand{\treepfva}{\\[2mm]\proof}
\newcommand{\treepfvb}{We need to show that $\bl T$ is a minimal spanning tree for $\bl G$. }
\newcommand{\treepfvc}{\\Since there are only finitely many edges and we keep adding edges and avoiding cycles,
  the algorithm will terminate and return a spanning tree~$\bl T$.}
\newcommand{\treepfvd}{\\We therefore only need to show that the tree $\bl T$ is minimal. }
\newcommand{\treepfve}{Assume that $\bl T$ is not minimal.
  \\In other words, $\bl w(S)\mlt w(T)$ for some minimal spanning tree~$\bl S$.}
\newcommand{\treepfvf}{\\Let $\bl s_1,\ldots,s_{n-1}$ and $\bl t_1,\ldots,t_{n-1}$ be the sequences of edges
  chosen by the algorithm to produce $\bl S$ and $\bl T$. }
\newcommand{\treepfvg}{Set $\bl i=\min\{j:w(s_j)\mlt w(t_j)\}$. }
\newcommand{\treepfvh}{\\Now, at least one edge $\bl s_j$ with $\bl 1\mleq j\mleq i$ lies outside $\bl \{t_1,\ldots,t_{i-1}\}$
  and can be added to it without making cycles. }
\newcommand{\treepfvj}{Also, $\bl w(s_j)\mleq w(s_i)\mlt w(t_i)$. }
\newcommand{\treepfvk}{However, this means that the algorithm would not have chosen $\bl t_i$ at the $\bl i$th step for $\bl T$, a contraction. }
\newcommand{\treepfvm}{Thus, $\bl T$ is a minimal spanning tree.\qed}

\newcommand{\treepfvia}{\\[2mm]\proof}
\newcommand{\treepfvib}{It is easy to see that $\bl T$ is a spanning tree. }
\newcommand{\treepfvic}{We must therefore show that the distances given by $\bl T$ from $\bl \mathsf{a}$
 to each vertex $\bl \mathsf{v}$ indeed equal $\bl \mathsf{d}(\mathsf{a},\mathsf{v})$ in $\bl G$,
 where, for any vertices $\bl \mathsf{w},\mathsf{x}\myin V(G)$,
 $\bl \textsf{d(w,x)}$ denotes the shortest distance between them. }
\newcommand{\treepfvid}{Assume that this is not true - that the algorithm fails, and let $\bl \mathsf{v}$ be the first vertex for which this happens. }
\newcommand{\treepfvidv}{Assume that the algorithm fails, and let $\bl \mathsf{v}$ be the first vertex for which this happens. }
\newcommand{\treepfvie}{Let $\bl T'$ be the tree generated up till this point,
  and let $\bl \mathsf{u}$ be the vertex preceding $\bl \mathsf{v}$ in $\bl T'$. }
\newcommand{\treepfvif}{Let $\bl P$ be a shortest path from $\bl \mathsf{a}$ to~$\bl \mathsf{v}$,
and let $\bl \mathsf{z}$ be the first vertex along this path that is not in $\bl T'$. }
\newcommand{\treepfvig}{\\Also, let $\bl \mathsf{y}$ be the preceding vertex in $\bl P$. }
\newcommand{\treepfvih}{Since the algorithm first failed with $\bl \mathsf{v}$,
 and $\bl \mathsf{v}$ was chosen to be in $\bl T'$ and $\bl \mathsf{z}$ was not,
 we have\vspace{-1mm}
  \[\bl\textsf{d(a,v)}\mlt \textsf{d(a,u)}+w(\textsf{u,v})\mleq\textsf{d(a,y)}+w(\textsf{y,z})\mlt \textsf{d(a,y)}+\textsf{d(y,v)}\meq\textsf{d(a,v)}\,,\]
\\[-6.5mm]a contradiction. \qed}
