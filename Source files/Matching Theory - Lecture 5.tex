\newcommand{\hallvab}{\\An {\dg alternating path} for $\bl M$ is a path where every second edge lies in $\bl M$.}
\newcommand{\hallvac}{\\An alternating path $\bl P$ is {\dg augmenting} if its end-edges lie outside $\bl M$.}
\newcommand{\hallvad}{\\Let $\bl E(P)$ denote the edges along $\bl P$.}
\newcommand{\hallvae}{{\darkgreen\sf Lemma}
  \\The set of edges $\bl {\darkgreen M'}= {\darkred M}\oplus {\blue E(P)}$ is a matching of size $\bl |M'| \meq |M| + 1$.}

\newcommand{\hallvpfia}{\\[2mm]\proof}
\newcommand{\hallvpfib}{\\The vertices covered by $\bl M'$ are those of $\bl P$. }
\newcommand{\hallvpfic}{\\Each is in one edge of $\bl M'$, so $\bl M'$ is a matching, and $\bl |M'|=|M|+1$.\qed\vspace*{-12mm}}

\newcommand{\hallvaf}{{\darkgreen\sf Theorem}
  \\$\bl \darkred M$ has maximal size in $\bl G$ if and only if $\bl \darkred M$ has no augmenting path.}

\newcommand{\hallvpfiia}{\\[2mm]\proof}
\newcommand{\hallvpfiib}{By the lemma, $\bl M$ is not maximal if an augmenting path exists.}
\newcommand{\hallvpfiic}{\\Suppose that no such path exists and let $\bl M'$ be a maximal matching in~$\bl G$.}
\newcommand{\hallvpfiid}{\\Each vertex of $\bl D \meq M\oplus M'$ is in at most one edge of $\bl M$ and of $\bl M'$.}
\newcommand{\hallvpfiie}{\\Thus, $\bl D$ is a disjoint union of cycles and alternating paths. }
\newcommand{\hallvpfiif}{\\The edges alternate between $\bl M$ and $\bl M'$, so each cycle has even edge~total. }
\newcommand{\hallvpfiig}{\\By assumption, no path is augmenting, so each path has even edge totals.}
\newcommand{\hallvpfiih}{\\Therefore, all of the cycles and paths contain the same number of edges from $\bl M$ as from $\bl M'$. }
\newcommand{\hallvpfiij}{This implies that $\bl |M-M'| \meq |M'-M|$.}
\newcommand{\hallvpfiik}{\\Hence, $\bl |M| \meq |M-M'|+|M\cap M'| \meq |M'-M|+|M\cap M'| \meq |M'|$.}
\newcommand{\hallvpfiim}{\\Thus, $\bl M$ has maximal size. \qed}

\newcommand{\hallvnf}{A {\dg flow} in $\bl D$ is a function $\bl \mathsf{\blue f}:\!A\rightarrow\mathbb{R}_{\mgeq 0}$ such that
  \\[1mm]\mybullet $\bl 0\mleq \blue\mathsf{f(a)\mleq c(a)}$ \:for each arc $\bl {\blue\mathsf a}\myin A$
  \\\mybullet $\bl \mathsf{\blue f^+(v)\meq f^-(v)}$   \:for each vertex $\bl {\blue\mathsf v}\myin V-\{\mathsf{s},\mathsf{t}\}$}
\newcommand{\hallvnfa}{\\[1mm]where\vspace{-7mm}
  \[{\blue\mathsf{f^-(v)}} \meq \sum_{\mathsf{u\,:\,(u,v)}\myin A}\mathsf{f(u,v)}
              \quad\mytext{and}\quad{\blue\mathsf{f^+(v)}} \meq \sum_{\mathsf{w\,:\,(v,w)}\myin A}\mathsf{f(v,w)}\]}
\newcommand{\hallvnfb}{\\[-4mm]The {\dg strength} of $\bl\mathsf{f}$ is $\bl \mathsf{\blue|f|\meq f^+(s)-f^-(s)}$.}

\newcommand{\hallvpa}{A {\dg cut} in $\bl D$ is a bipartition $\bl (X,Y)$ of $\bl V$
  with $\bl {\blue\mathsf s}\myin X$ and $\bl {\blue\mathsf t}\myin Y$.}
\newcommand{\hallvpb}{\\The {\dg capacity} of a cut $\bl(X,Y)$ is the sum\vspace{-1.5mm}
  \[{\blue\mathsf{c}(X,Y)} \meq \sum_{\mathsf{u}\myin X,\,\mathsf{v}\myin Y\::\mathsf{(u,v)}\myin A} \mathsf{c(u,v)}\]}

\newcommand{\hallvra}{{\darkgreen\sf Theorem}
  \\For each flow $\bl \mathsf{f}$ and each cut {\blue $\bl (X,Y)$} on $\bl D$, we have $\bl|\mathsf{f}|\mleq \mathsf{c}(X,Y)$.}
\newcommand{\hallvrav}{{\darkgreen\sf Theorem}
  \\$\blue|\mathsf{f}|\mleq \mathsf{c}(X,Y)$.}

\newcommand{\hallvrb}{Let $\bl (X,Y)$ be a partition of $\bl D$ and $\bl\mathsf{f}$ be a flow on $\bl D$, and define\vspace{-1mm}
  \[{\blue\mathsf{f}(X,Y)} \meq \sum_{\mathsf{u}\myin X,\,\mathsf{v}\myin Y\::\mathsf{(u,v)}\myin A} \mathsf{f(u,v)}\]}

\newcommand{\hallvrc}{{\darkgreen\sf Theorem}
  \\If $\bl(X,Y)$ is a cut of $\bl D$, then $\bl|\mathsf{f}|\meq \mathsf{f}(X,Y)-\mathsf{f}(Y,X)$.}

\newcommand{\hallvpfiiia}{\\[3mm]\proof}
\newcommand{\hallvpfiiib}{\\By definition, $\bl \mathsf{f^+(v)-f^-(v)}=\mathsf0$ for all $\bl \mathsf{v}\myin X-\{\mathsf{s}\}$. }
\newcommand{\hallvpfiiic}[4]{Therefore,\vspace{-1mm}
  \begin{align*}\bl
         \mathsf{|f|}
   &\meq \mathsf{f^+(s)} - \mathsf{f^-(s)}\\\bl
   &\hspace{1.2mm}{#1{\meq\sum_{\mathsf{u}\myin X}\mathsf{f^+(u)} - \sum_{\mathsf{v}\myin X}\mathsf{f^-(v)}}}\\\bl
   &\:\,{#2{\meq\sum_{\mathsf{u}\myin X}\sum_{\mathsf{v:(u,v)}\myin A}\mathsf{f(u,v)}
        -\sum_{\mathsf{v}\myin X}\sum_{\mathsf{u:(u,v)}\myin A}\mathsf{f(u,v)}}}\\\bl
   &\:\,{#3{\meq\sum_{\mathsf{u},\mathsf{v}\myin X:\atop\mathsf{(u,v)}\myin A}\mathsf{f(u,v)}
        +\sum_{\mathsf{u}\myin X,\mathsf{v}\myin Y:\atop\mathsf{(u,v)}\myin A}\mathsf{f(u,v)}
        -\sum_{\mathsf{v},\mathsf{u}\myin X:\atop\mathsf{(u,v)}\myin A}\mathsf{f(u,v)}
        -\sum_{\mathsf{v}\myin X,\mathsf{u}\myin Y:\atop\mathsf{(u,v)}\myin A}\mathsf{f(u,v)}}}\\\bl
   &\:\,{#4{\meq\mathsf{f}(X,Y)-\mathsf{f}(Y,X)\qquad\black\square}}\\[-8mm]
  \end{align*}}

\newcommand{\hallvpfiva}{\\[3mm]\proof}
\newcommand{\hallvpfivb}{\\$\bl|\mathsf{f}|= \mathsf{f}(X,Y)-\mathsf{f}(Y,X)\mleq\mathsf{f}(X,Y)\mleq\mathsf{c}(X,Y)$.\qed}

\newcommand{\hallvrd}{{\darkgreen Corollary}
  \\$\blue|\mathsf{f}|\meq \mathsf{f^-(t)-f^+(t)}$.}

\newcommand{\hallvpfva}{\\[3mm]\proof}
\newcommand{\hallvpfvb}{\\$\bl|\mathsf{f}|\meq \mathsf{f}(V-\{t\},\{t\})-\mathsf{f}(\{t\},V-\{t\}) \meq \mathsf{f^-(t)}-\mathsf{f^+(t)}$.\qed}

\newcommand{\hallvre}{Let $\bl\mathsf{f}$ be a flow on the network $\bl {\blue D} \meq (V,A)$. }

\newcommand{\hallvrf}{\\An {\dg augmenting path} $\bl P$ for $\bl\mathsf{f}$ is
  a sequence of vertices $\bl {\blue\mathsf{v_0,v_1,\ldots,v_n}}\myin V$ for which either
  \\[2mm]\mybullet $\bl e \meq (\mathsf{v_i},\mathsf{v_{i+1}})\myin A$ and $\bl {\blue\alpha_i} \meq \mathsf{c}(e)-\mathsf{f}(e)\mgt 0$; or
       \\\mybullet $\bl e \meq (\mathsf{v_{i+1}},\mathsf{v_i})\myin A$ and $\bl {\blue\alpha_i} \meq \mathsf{f}(e)\mgt 0$.}

\newcommand{\hallvrg}{Define $\bl {\blue\alpha} \meq \min\{\alpha_i\}$ and let $\bl\mathsf{f}\oplus P$ denote
  the function given on $\bl A$ by \vspace{-1mm}
  \[\blue(\mathsf{f}\oplus P)(e) \meq \begin{cases}
      \mathsf{f}(e)+\alpha & \:\mytext{if }\: e \meq (\mathsf{v_i},\mathsf{v_{i+1}})\myin A\\
      \mathsf{f}(e)-\alpha & \:\mytext{if }\: e \meq (\mathsf{v_{i+1}},\mathsf{v_i})\myin A\end{cases}\]}

\newcommand{\hallvrh}{{\darkgreen\sf Theorem}
  \\If   $\bl\mathsf{f}$ has an augmenting path $\bl P$ from $\bl\mathsf{s}$ to $\bl\mathsf{t}$,
  \\then $\bl\mathsf{f}\oplus P$ is a flow with $\bl|\mathsf{f}\oplus P|\mgt |\mathsf{f}|$.}

\newcommand{\hallvpfvia}{\\[3mm]\proof}
\newcommand{\hallvpfvib}{\hfill It is easy to check that $\bl \mathsf{f}\oplus P$ satisfies the two flow conditions.\\}
\newcommand{\hallvpfvic}{Furthermore, $\bl |\mathsf{f}\oplus P|\meq|\mathsf{f}|+\alpha\mgt |\mathsf{f}|$ where $\bl \alpha$ is defined as above.\qed}

\newcommand{\hallvrj}{{\darkgreen Corollary}
  \\If $\bl\mathsf{f}$ is a maximal flow (i.e., has maximal strength $\bl |\mathsf{f}|$),
  \\then there is no augmenting path for $\bl\mathsf{f}$ from $\bl\mathsf{s}$ to $\bl\mathsf{t}$.}

\newcommand{\hallvsa}{{\darkgreen Max-Flow Min-Cut Theorem} {\gray (Ford and Fulkerson 1955)}
  \\In each network, maximal flow strength {\dg equals} minimal cut capacity:
  $\bl\max|\mathsf{f}|\meq\min\mathsf{c}(X,Y)$.}

\newcommand{\hallvsaa}{\\[3mm]{\gray This theorem is also called the {\green Ford-Fulkerson Theorem}.}}

\newcommand{\hallvpfviia}{\\[3mm]\proof\quad}
\newcommand{\hallvpfviib}{Let any $\bl \mathsf{f}$ be a maximal flow in $\bl D$.}
\newcommand{\hallvpfviic}{\\Let $\bl S$ be all vertices reachable by
  an augmenting path for $\bl \mathsf{f}$ starting in $\bl \mathsf{s}$. }
\newcommand{\hallvpfviid}{\\By the previous corollary, $\bl \mathsf{f}$ has no augmenting path from $\bl \mathsf{s}$ to $\bl \mathsf{t}$.}
\newcommand{\hallvpfviie}{\\Thus, $\bl \mathsf{t}\myin T$ where $\bl T\meq V-S$, so $\bl (S,T)$ is a cut. }
\newcommand{\hallvpfviif}{Let $\bl \mathsf{u}\myin S$ and $\bl \mathsf{v}\myin T$.}
\newcommand{\hallvpfviig}{\\Suppose that $\bl e\meq(\mathsf{u},\mathsf{v})\myin A$.}
\newcommand{\hallvpfviih}{\\Since $\bl \mathsf{v}$ cannot extend any augmenting path ending in $\bl \mathsf{u}$,
  $\bl \mathsf{f}(e)\meq\mathsf{c}(e)$. }
\newcommand{\hallvpfviij}{\\Suppose then that $\bl e \meq (\mathsf{v},\mathsf{u})\myin A$. }
\newcommand{\hallvpfviik}{\\Since $\bl \mathsf{v}$ cannot extend any augmenting path ending in $\bl \mathsf{u}$,
  $\bl \mathsf{f}(e)\meq0$. }
\newcommand{\hallvpfviim}{\\Hence, by a previous theorem,\vspace{-1mm}
  \[\bl  |\mathsf{f}|
    \meq \mathsf{f}(S,T)-\mathsf{f}(T,S)
    \meq \hspace{-2mm}\sum_{\mathsf{u}\myin S,\mathsf{v}\myin T\::\atop\mathsf{(u,v)}\myin A} \hspace{-3mm}\mathsf{f(u,v)}
     -\hspace{-3mm}\sum_{\mathsf{u}\myin S,\mathsf{v}\myin T\::\atop\mathsf{(v,u)}\myin A} \hspace{-3mm}\mathsf{f(u,v)}
    \meq \hspace{-3mm}\sum_{\mathsf{u}\myin S,\mathsf{v}\myin T\::\atop\mathsf{(u,v)}\myin A} \hspace{-3mm}\mathsf{c(u,v)}
    \meq \mathsf{c}(S,T)\]}
\newcommand{\hallvpfviimv}{\\Hence, by a previous theorem, $\bl |\mathsf{f}| \meq \cdots \meq \mathsf{c}(S,T)$.
  \\Hence, $\bl \min \mathsf{c}(X,Y)\mleq \mathsf{c}(S,T)\meq|\mathsf{f}|$. }
\newcommand{\hallvpfviin}{\\[-4mm]Hence, $\bl \min \mathsf{c}(X,Y)\mleq \mathsf{c}(S,T)\mleq |\mathsf{f}|$. }
\newcommand{\hallvpfviip}{\\However, we have previously shown that $\bl |\mathsf{f}|\mleq \mathsf{c}(X,Y)$ for any cut~$\bl (X,Y)$. }
\newcommand{\hallvpfviiq}{\\Thus, $\bl |\mathsf{f}|\meq \min \mathsf{c}(X,Y)$.\qed}

\newcommand{\hallvtbb}{\\\mybullet Search for an augmenting path $\bl P$ for $\bl \mathsf{f}$ from $\bl\mathsf{s}$ to $\bl\mathsf{t}$.}
\newcommand{\hallvtbc}{\\\mybullet Replace $\bl \mathsf{f}$ by $\bl \mathsf{f}\oplus P$.}
\newcommand{\hallvtbd}{\\\mybullet Continue until there is no augmenting path.}
\newcommand{\hallvtbe}{\\\mybullet Then $\bl \mathsf{f}$ is a maximal flow.}
\newcommand{\hallvtbav}{\\[1mm]\mystep Set $\bl\mathsf{f} \meq \mathsf{0}$.}
\newcommand{\hallvtbbv}{\\\mystep Search for an augmenting path $\bl P$ for $\bl\mathsf{f}$ from $\bl\mathsf{s}$ to $\bl\mathsf{t}$.}
\newcommand{\hallvtbcv}{\\\mystep Replace $\bl\mathsf{f}$ by $\bl\mathsf{f}\oplus P$.}
\newcommand{\hallvtbdv}{\\\mystep Continue until there is no augmenting path.}
\newcommand{\hallvtbev}{\\\mystep Then $\bl\mathsf{f}$ is a maximal flow.}

\newcommand{\halliiae}{{\darkgreen K\"onig's Theorem} {\gray (1916)}
   \\The maximal size $\bl M$ of a partial transversal in a {\bl(0,1)}-matrix $\bl N$ {\dg equals}
   \\the minimal number $\bl m$ of lines needed to cover all {\bl1}-entries of $\bl N$.}


\newcommand{\hallvpfixa}{\\[3mm]\proof\quad}
\newcommand{\hallvpfixb}{\\Let $\bl R$ and $\bl C$ be the row and column indices of $\bl N$. }
\newcommand{\hallvpfixc}{\\Define network $\bl D=(V,A)$
  where $\bl V \meq R\cup C \cup\{\mathsf{s},\mathsf{t}\}$ and $\bl A \meq S\cup B\cup T$ for
  $\bl S \meq \{\mathsf{s}\}\times R$,
  $\bl B \meq \{(i,j)\myin R\times C:n_{ij}\meq 1\}$, and
  $\bl T \meq C\times \{\mathsf{t}\}$,
  with capacities $\bl \mathsf{c}(e) \meq M+1$ if $\bl e\myin B$ and $\bl \mathsf{c}(e) \meq 1$ otherwise. }
\newcommand{\hallvpfixcv}{Define the network $\bl D \meq (V,A)$ where $\bl V \meq R\cup C \cup\{\mathsf{s},\mathsf{t}\}$ and $\bl A \meq S\cup B\cup T$ for
  \begin{align}
    S &= \{(\mathsf{s},\mathsf{i})\::i\myin R\}\\
    B &= \{(\mathsf{i},\mathsf{j})\myin R\times C\::n_{ij}=1\}\\
    T &= \{(\mathsf{j},\mathsf{t})\::j\myin C\}\end{align}with capacities\[
    \mathsf{c}(e) \meq \begin{cases} 1  & \:\mytext{if }\, e \meq (\mathsf{s},\mathsf{i}) \textsf{ or } (\mathsf{j},\mathsf{t}) \textsf{ for some }i,j\myin R\cup C\\
                                 M+1 & \:\mytext{if }\, e \meq (\mathsf{i},\mathsf{j}) \textsf{ for some }\mathsf{i},\mathsf{j}\myin R\cup C\end{cases}\]}
\newcommand{\hallvpfixd}{\\Then there is a maximal flow $\bl \mathsf{f}$ with $\bl \mathsf{f}(e)\myin \{0,1\}$ for all edges $\bl e\myin A$.}
\newcommand{\hallvpfixe}{\\Therefore, the arcs $\bl e\myin B$ with positive flow ($\bl\mathsf{f}(e)\meq1$) are
  vertex-disjoint and correspond to a partial transversal of $\bl A$.}
\newcommand{\hallvpfixf}{\\Hence, by the {\dg Max-Flow Min-Cut Theorem}, $\bl M\mgeq \min \mathsf{c}(X,Y)$. }
\newcommand{\hallvpfixg}{\\Let $\bl (X,Y)$ be a minimal edge cut. }
\newcommand{\hallvpfixh}{Since $\bl \mathsf{c}(X,Y)\meq\max|\mathsf{f}|\mlt M+1$,
  \\no edge of $\bl B$ is cut by $\bl (X,Y)$}
\newcommand{\hallvpfixj}{, so $\bl \mathsf{c}(X,Y)\meq|U|$,
  \\where $\bl U \meq (R\cap Y)\cup (C\cap X)$ is a set of vertices touching all edges of~$\bl B$.}
\newcommand{\hallvpfixk}{\\Thus, $\bl U$ corresponds in $\bl N$ to a set of lines covering all {\bl1}-entries. }
\newcommand{\hallvpfixm}[1]{\\Hence, $\bl m\mleq |U| \meq \mathsf{c}(X,Y)\mleq M$.
 {#1{Since $\bl m\mgeq M$, we have equality.\qed}}\vspace*{-12mm}}
