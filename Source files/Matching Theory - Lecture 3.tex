\newcommand{\halliiipfie}{\\Then each entry of $\bl M'$ is nonnegative and each line sum equals $\bl t+|a|n$. }

\newcommand{\halliiipfig}{\\If $\bl t=0$, then $\bl M \meq 0\times P$ for any permutation matrix $\bl P$, so assume $\bl t\mgt 0$.}
\newcommand{\halliiipfih}{\\We use induction on the number of nonzero entries in~$\bl M$.}
\newcommand{\halliiipfij}{\\Define $\bl \mathcal{A}\meq\{A_i\}_I$
  where $\bl A_i \meq \{j\::\:m_{ij}\mgt 0\}$ and $\bl I\meq\{1,\dots,n\}$.}
\newcommand{\halliiipfik}{\\Consider any subset $\bl J\msubseteq I$. }
\newcommand{\halliiipfim}{Then\vspace{-1.5mm}
  \[\bl|\mathcal{A}(J)|t \meq \sum_{j\myin\mathcal{A}(J)}\sum_{i\myin I} m_{ij}
      \mgeq \sum_{j\myin\mathcal{A}(J)}\sum_{i\myin J}m_{ij}
      \mgeq  \sum_{i\myin J}\sum_{j\myin A_i}m_{ij}
      \meq  \sum_{i\myin J}t

\newcommand{\halliiipfin}{\\[-3mm]Hence, $\bl |\mathcal{A}(J)|\mgeq |J|$. }
\newcommand{\halliiipfio}{By {\dg Hall's Theorem}, $\bl \mathcal{A}$ has a matching,

\newcommand{\halliiipfir}{\\also, $\bl M'$ has fewer nonzero entries than~$\bl M$.}
\newcommand{\halliiipfis}{\\By induction, $\bl M'=\sum c_iP_i$ for permutation matrices $\bl P_i$ and $\bl \sum c_i \meq t-c$. }

\newcommand{\halliiipfiia}{\\[2.5mm]\proof}
\newcommand{\halliiipfiib}{\\By induction, we need only show that one row can be added. }
\newcommand{\halliiipfiic}{\\Let $\bl A_i$ be the set of elements {\red not} in the $\bl i$th column of the rectangle~$\bl R$,
  \\and set $\bl \mathcal{A}\meq \{A_i\}_I$ where $\bl I\meq\{1,\dots,n\}$. \\Note that $\bl |A_i| \meq n - r$. }
\newcommand{\halliiipfiid}{\\Consider any subset $\bl J\msubseteq I$. }
\newcommand{\halliiipfiie}{\\Each element appears $\bl r$ times in $\bl R$
                           \\and therefore $\bl n-r$ times in the sets~$\bl A_i$.}
\newcommand{\halliiipfiif}{\\Therefore, each element can appear at most
                           \\$\bl n-r$ times in the sets $\bl A_i$ where $\bl i\myin J$. }
\newcommand{\halliiipfiig}{\\Then\vspace{-2.5mm}
  \[\bl|J|(n-r) \meq \sum_{i\myin J}|A_i| \meq \sum_{j\myin\mathcal{A}(J)}|\{i\myin J\::\: j\myin A_i\}|\mleq |\mathcal{A}(J)|(n-r)\,\black.\vspace*{-1mm}\]}

\newcommand{\halliiihc}{{\dg Sperner's Theorem} {\gray(1928)}
  \\If $\bl\mathcal{A}$ is a family of subsets of $\bl \{1,\ldots,n\}$ with $\bl A\not\msubset B$ for all $\bl A,B\myin\mathcal{A}$,
  %  so that no member is a subset of another,
  then $\bl|\mathcal{A}|\mleq\binom{n}{\lfloor\frac{n}{2}\rfloor}$.}

\newcommand{\halliiihd}[1]{\mypicture{\rput(1.5,-4.6){\psset{unit=9mm,dotscale=1.5,linewidth=.03,
  fillcolor=halfgray,linecolor=gray,shadowcolor=offwhite,shadow=true,shadowsize=.125,shadowangle=-45}
  \pspicture(-.7,1)(6,3.7)#1\endpspicture}}\\[-5.2mm]}

\newcommand{\halliiipfiiia}{\\[3.5mm]\proof}
\newcommand{\halliiipfiiib}{\\The family $\bl \mathcal{A}$ is an antichain in $\bl \mathcal{B}_n$. }
\newcommand{\halliiipfiiic}{By the trivial direction of {\dg Dilworth's Theorem}, 
 $\bl|\mathcal{A}|\mleq m$
  where $\bl m$ is the minimal number of chains to cover~$\bl \mathcal{B}_n$. }
\newcommand{\halliiipfiiid}{\\The {\dg Symmetric Chain Decomposition Algorithm} gives a covering of $\bl \mathcal{B}_n$
  by disjoint chains that each contain one member $\bl P$ of size $\bl |P|\meq\lfloor\frac{n}{2}\rfloor$.}
\newcommand{\halliiipfiiie}{\\There are $\bl \binom{n}{\lfloor\frac{n}{2}\rfloor}$ such members.}
\newcommand{\halliiipfiiif}{\\Hence, $\bl |\mathcal{A}|\mleq m\mleq\binom{n}{\lfloor\frac{n}{2}\rfloor}$. }
\newcommand{\halliiipfiiig}{\\(Indeed, $\bl m \meq \binom{n}{\lfloor\frac{n}{2}\rfloor}$.) }

\newcommand{\halliiihe}{{\dg The Erd\H{o}s-Szemeredi Theorem} {\gray (1935)}
  \\Each sequence $\bl a_1,\ldots,a_{n^2\!+\!1}$ of $\bl n^2\!+\!1$ distinct integers has
  \\a subsequence with $\bl n+1$ elements that is either increasing or decreasing.}

  \\Consider the following sequence of $\bl n^2\!+\!1$ integers where $\bl n\meq\textsf{3}$:}

  \\where $\bl a_i\preceq a_j$ if and only if $\bl i\mleq j$ and $\bl a_i\mleq a_j$.}
\newcommand{\halliiipfivc}{\\The chains in $\bl P$ correspond to the increasing subsequences of $\bl a_1,\ldots,a_{n^2\!+\!1}$
  \\and the antichains correspond to the decreasing subsequences. }
\newcommand{\halliiipfivd}{\\If there is an increasing subsequence of size $\bl n+1$, then we are done.}
\newcommand{\halliiipfive}{\\Suppose then that all increasing subsequences have length at most $\bl n$. }
\newcommand{\halliiipfivf}{\\In other words, the chains of $\bl P$ have length at most $\bl n$. }
\newcommand{\halliiipfivg}{\\By the {\dg Dual Dilworth's Theorem}, we can cover $\bl P$ by at most $\bl n$ antichains. }
\newcommand{\halliiipfivh}{\\Since $\bl |P|\mgt n^2$, at least one antichain must have at least $\bl n+1$ elements. }
\newcommand{\halliiipfivj}{\\Thus, there is a decreasing subsequence of size at least $\bl n+1$.\qed}
